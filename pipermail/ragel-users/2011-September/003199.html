Hi Shawn, does this work for you?

A = (
    start: A: (
        IDLE -> A |
        POLL @poll OPEN -> A |
        POLL @poll CLOSED -> B
    ),
    B: (IDLE -> B | EXIT -> final)
) >a_entry;

m := START A;

-Adrian
